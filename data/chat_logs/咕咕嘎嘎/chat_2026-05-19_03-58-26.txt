# include <vector>
using namespace std;

struct TicTacToe {
    vector<vector<char>> board = {
        {'+','+','+'},
        {'+','+','+'},
        {'+','+','+'}
    };
    char current_player = 'X';

    // 显示棋盘
    void show_board() {
        for (int i = 0; i < board.size(); i++) {
            for (int j = 0; j < board[i].size(); j++) {
                cout << board[i][j] << "    ";
            }
            cout << endl;
        }
    }

    // 检查能不能下
    bool check_ok(int &x, int &y) {
        return board[x][y] == '+';
    }

        cout << "其他的说明" << endl;
        while (true) {
            cout << "请" << current_player << "落子，空格隔开：";
            cin >> x >> y;

            // 输入失败
            if (cin.fail()) {
                cin.clear();  // 清除错误状态
                cin.ignore(10000);  // 丢弃错误的输入，10000是字符数量，可以有换行参数'\n'
                cout << "输入错误，需要两个整数" << endl;

            if (x < 0 || x > board.size()-1 || y < 0 || y > board[0].size()-1) {
                cout <<  "坐标必须在 0 到 2 之间" << endl;
            } else if (!check_ok(x, y)) {
                cout << "该位置已被占用，请重新输入" << endl;
            } else {
                return;
            }
        }
    }

    // 落子
    void move(int &x, int &y) {
        board[x][y] = current_player;
    }

    // 判断这一步是否可以胜利

        char p = board[x][y];

        // 检查x行, y列
        if (board[x][0] == p && board[x][1] == p && board[x][2] == p) return true;
        if (board[0][y] == p && board[1][y] == p && board[2][y] == p) return true;
        if (x == y) {
            if (board[0][0] == p && board[1][1] == p && board[2][2] == p) return true;
        }
        if (x + y == 2) {
            if (board[0][2] == p && board[1][1] == p && board[2][0] == p) return true;
        }

        return false;
    }

    // 判断棋盘是否满了
    bool is_full() {

                if (board[i][j] == '+') {
                    return false;
                }
            }

    }
};
